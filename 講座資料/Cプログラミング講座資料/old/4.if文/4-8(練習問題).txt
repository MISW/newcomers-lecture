1問目 ユーザーから1つの整数を入力してもらい、30より大きいか、それ以下か判定するプログラム

2問目 ユーザーから2つの整数を入力してもらい、その2つを掛けた数は50より大きいかそれ以下かの判定プログラム

3問目 1〜100までの数を表示する、しかし
　　　3の倍数の時はFizz
      5の倍数の時はBuzz
      3の倍数かつ5の倍数の時はFizzBuzzと表示されるプログラム
例

2

4
Buzz
Fizz
7
8
Fizz
Buzz

Fizz

解答

1問目
#include<stdio.h>

int main(){
	
	int a;
	
	printf("数を１つ入力して下さい。\n");
	scanf("%d",&a);
	
	if(a>30){
		printf("%d は 30より大きいです。\n");
	}
	else{
		printf("%d は30以下です。\n");
	}
	
	return 0;
}

2問目

#include<stdio.h>

int main(){
	
	int a,b,ans;
	
	printf("1つ目の数を入力して下さい。\n");
	scanf("%d",&a);
	printf("2つ目の数を入力して下さい。\n");
	scanf("%d",&b);
	
	ans = a*b;
	
	if(ans > 50){
		printf("%d * %dは50よりも大きいです。\n",a,b);
	}
	else{
		printf("%d * %dは50以下です。\n",a,b);
	}
	
	return 0;
}

問3

#include<stdio.h>

int main(){
	
	int i;
	
	for(i=1;i<=100;i++){
		
		if(i%15==0){
			printf("FizzBuzz\n");
		}
		else if(i%3==0){
			printf("Fizz\n");
		}
		else if(i%5==0){
			printf("Buzz\n");
		}
		else{
			printf("%d\n",i);
		}
	}
	
	return 0;
}
