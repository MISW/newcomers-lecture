#include<stdio.h>

void tenHelloWorld(){
	
	int i;
	
	for(i=0;i<10;i++){
		
		printf("Hello World\n");
		
	}
	printf("\n");
	
	return 0;
}

int main(){
	
	int i;
	for(i=0;i<10;i++){
		
		tenHelloWorld();
		
	}
	
	return 0;
}

/*以下解説。

tenHelloWorld()を少しだけ変更
（最後にprintf("\n");によって改行。)

main関数内をみてわかるように、
自分で作った関数は何度でも使用可能です。

*/
