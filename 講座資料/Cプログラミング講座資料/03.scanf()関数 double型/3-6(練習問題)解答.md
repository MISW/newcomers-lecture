#### 1問目
```
#include<stdio.h>

int main(){
	
	double r,l,s;
	
	printf("円の半径を入力してください\n");
	scanf("%lf",&r);
	
	l = 2 * 3.14 * r;
	s = 3.14 * r * r;
	
	printf("円の周の長さは %f\n",l);
	printf("円の面積は %f",s);
	
	return 0;

}
```

#### 2問目
```
int main(){
	
	int n,m,i,ans=1;
	
	printf("n^mを計算します\n");
	printf("nを入力してください");
	scanf("%d",&n);
	
	printf("mを入力してください");
	scanf("%d",&m);
	
	for(i=0;i<m;i++){
		ans *= n;
	}
	printf("%d ^ %d = %d",n,m,ans);
	
	return 0;
}
```
